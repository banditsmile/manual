<script type="text/javascript">
var msg = function(message) {
$('#msg').html(message + '<br />' + $('#msg').html());
};
var method_msg = function(message) {
$('#method_msg').html(message + '<br />' + $('#method_msg').html());
};
var method_destroy = function() {
$('#woman_clothing').sortable('destroy');
method_msg('sortable组件被销毁, 所有元素恢复最初的DOM结构.');
};
var method_disable = function() {
$('#woman_clothing').sortable('disable');
method_msg('设置失效');
};
var method_enable = function() {
$('#woman_clothing').sortable('enable');
method_msg('设置生效');
};
var method_option = function() {
var name = $('#option_name').val();
var value = $('#option_value').val();
if(!name) return ;
if(value) $('#woman_clothing').sortable('option', name, value);
else $('#option_value').val($('#woman_clothing').sortable('option', name));
method_msg('设置或查看选项[<font color="blue">' + name + '</font>]的值成功.');
};
var method_serialize = function() {
var result = $('#woman_clothing').sortable('serialize', {
connected: true,
attribute: 'sn',
key: 'id',
expression: /^(?:clothing_)(.+)$/
});
method_msg('序列化成功, 序列化结果: [<font color="red">' + result + '</font>]');
};
var method_toArray = function() {
var result = $('#woman_clothing').sortable('toArray', {
connected: true,
attribute: 'sn'
});
method_msg('数组转换成功, 序列化结果: [<font color="red">"' + result.join('", "') + '"</font>]');
};
var method_refresh = function() {
$('#woman_clothing').sortable('refresh');
method_msg('刷新成功.');
};
var method_refreshPositions = function() {
$('#woman_clothing').sortable('refreshPositions');
method_msg('刷新位置成功.');
};
var method_cancel = function() {
try {
$('#woman_clothing').sortable('cancel');
method_msg('取消上一次排序成功.');
} catch(e) {
method_msg('由于还没有过排序, 所以不能取消.');
}
};
$(function() {
var $tabs = $('.tabs').tabs();
var $tab_items = $('ul:first li', $tabs).droppable({
accept: '.commodity',
hoverClass: 'ui-state-hover',
tolerance: 'pointer',
over: function(event, ui) {
var $item = $(this);
var $list = $($item.find('a').attr('href'));
$('.tabs').tabs('select', $tab_items.index($item));
}
});
$('#woman_clothing, #sport, #cosmetic, #dummy').sortable({
cursor: 'move',
items: '.commodity',
cancel: '.operate-area',
handle: '>img',
placeholder: 'placeholder',
connectWith: '#woman_clothing, #sport, #cosmetic, #dummy',
tolerance: 'pointer',
revert: true,
opacity: 0.5,
start: function(event, ui) {
msg('事件[' + event.type + ']: <font color="blue">' + ui.item.text() + '</font>开始拖动.');
},
/*sort: function(event, ui) {
msg('事件[' + event.type + ']: <font color="blue">' + ui.item.text() + '</font>拖动中, 当前坐标left: ' + ui.offset.left + ', top: ' + ui.offset.top + '.');
}, */
change: function(event, ui) {
msg('事件[' + event.type + ']: <font color="blue">' + ui.item.text() + '</font>一次有效的顺序改变.');
},
beforeStop: function(event, ui) {
msg('事件[' + event.type + ']: <font color="blue">' + ui.item.text() + '</font>准备停止.');
},
stop: function(event, ui) {
msg('事件[' + event.type + ']: <font color="blue">' + ui.item.text() + '</font>停止.');
},
update: function(event, ui) {
msg('事件[' + event.type + ']: <font color="blue">' + ui.item.text() + '</font>停止排序并且顺序被真正改变了.');
},
receive: function(event, ui) {
msg('事件[' + event.type + ']: <font color="blue">' + ui.item.parent().attr('text') + '</font>从<font color="blue">' + ui.sender.attr('text') + '</font>接受到<font color="blue">' + ui.item.text() + '</font>.');
},
remove: function(event, ui) {
msg('事件[' + event.type + ']: <font color="blue">' + ui.item.text() + '</font>从' + ui.item.parent().attr('text') + '移除了.');
},
over: function(event, ui) {
msg('事件[' + event.type + ']: <font color="blue">' + ui.item.text() + '</font>到可以放入的另一个列表了.');
},
out: function(event, ui) {
msg('事件[' + event.type + ']: <font color="blue">' + ui.item.text() + '</font>离开原来的列表<font color="blue">' + ui.item.parent().attr('text') + '</font>了.');
},
activate: function(event, ui) {
msg('事件[' + event.type + ']: 一个可以放入<font color="red">' + $(this).attr('text') + '</font>的项开始活动了.');
},
deactivate: function(event, ui) {
msg('事件[' + event.type + ']: 一个可以放入<font color="red">' + $(this).attr('text') + '</font>的项停止活动了.');
}
});
});
